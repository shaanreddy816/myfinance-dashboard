## 🧪 Testing Examples

### Example 1: Create and Validate Master Expense (EMI)

```javascript
// Create a home loan EMI
const homeLoan = new MasterExpense({
    family_id: 'user@example.com',
    created_by: 'user@example.com',
    name: 'Home Loan EMI',
    amount: 35000,
    category: 'housing',
    expense_type: 'emi',
    is_emi: true,
    emi_type: 'house_loan',
    start_date: '2024-01-01',
    end_date: '2044-01-01'
});

// Calculate tenure and total
homeLoan.tenure_months = homeLoan.calculateTenure(); // 240 months
homeLoan.total_amount = homeLoan.calculateTotalAmount(); // ₹8,400,000

// Validate
const validation = homeLoan.validate();
console.log(validation); // { valid: true, errors: [] }

// Calculate remaining
console.log(homeLoan.calculateRemainingTenure()); // e.g., 228 months
console.log(homeLoan.calculateRemainingAmount()); // e.g., ₹7,980,000
```

### Example 2: Create and Validate Individual Expense

```javascript
// Create a manual expense
const movieNight = new ExpenseEntry({
    family_id: 'user@example.com',
    created_by: 'user@example.com',
    amount: 2500,
    category: 'entertainment',
    description: 'Movie tickets and dinner',
    entry_date: '2024-12-01',

```

### Example 3: Auto-Generated Expense with Manual Adjustment

```javascript
// Create auto-generated expense
const groceries = new ExpenseEntry({
    family_id: 'user@example.com',
    master_expense_id: 'master-uuid-123',
    amount: 15000,
    category: 'groceries',
    description: 'Monthly Groceries (Auto-generated)',
    entry_date: '2024-12-01',
    auto_generated: true,
    created_by: 'system'

// User adjusts the amount

console.log(groceries.original_amount); // 15000
console.log(groceries.manually_adjusted); // true
console.log(groceries.isManuallyAdjusted()); // true
```
